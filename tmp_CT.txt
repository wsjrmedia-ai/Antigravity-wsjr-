//+------------------------------------------------------------------+
//|                        Master.mq5                                |
//|  Master EA: Writes open positions to file for slave accounts     |
//+------------------------------------------------------------------+
#property strict
#property version "1.00"

#include <Trade/Trade.mqh>

input string CopyChannel = "DEFAULT"; // Must match slave
input int SyncSeconds  = 1;         // How often to export data

//--- structure for exporting
struct MasterPos
{
   ulong  ticket;
   string symbol;
   int    type;
   double volume;
   double open_price;
   double sl;
   double tp;
   double equity;
   long   magic;
   string comment;
};

CTrade trade;

//+------------------------------------------------------------------+
int OnInit()
{
   EventSetTimer(SyncSeconds);
   return(INIT_SUCCEEDED);
}
//+------------------------------------------------------------------+
void OnDeinit(const int reason)
{
   EventKillTimer();
}
//+------------------------------------------------------------------+
void OnTimer()
{
   ExportMasterPositions();
}
//+------------------------------------------------------------------+
void ExportMasterPositions()
{
   string file = "CopyMaster_" + CopyChannel + ".csv";

   int handle = FileOpen(file,
                         FILE_COMMON | FILE_WRITE | FILE_CSV | FILE_ANSI);

   if(handle == INVALID_HANDLE)
   {
      Print("Master: Error opening file! Code=", GetLastError());
      return;
   }

   //--- Write header
   FileWrite(handle,
             "ticket","symbol","type","volume","open_price",
             "sl","tp","equity","magic","comment");

   double eq = AccountInfoDouble(ACCOUNT_EQUITY);

   int total = PositionsTotal();

   for(int i = 0; i < total; i++)
   {
      ulong ticket = PositionGetTicket(i);
      if(!PositionSelectByTicket(ticket))
         continue;

      MasterPos mp;
      mp.ticket      = ticket;
      mp.symbol      = PositionGetString(POSITION_SYMBOL);
      mp.type        = (int)PositionGetInteger(POSITION_TYPE);
      mp.volume      = PositionGetDouble(POSITION_VOLUME);
      mp.open_price  = PositionGetDouble(POSITION_PRICE_OPEN);
      mp.sl          = PositionGetDouble(POSITION_SL);
      mp.tp          = PositionGetDouble(POSITION_TP);
      mp.equity      = eq;
      mp.magic       = PositionGetInteger(POSITION_MAGIC);
      mp.comment     = PositionGetString(POSITION_COMMENT);

      FileWrite(handle,
                mp.ticket, mp.symbol, mp.type, mp.volume, mp.open_price,
                mp.sl, mp.tp, mp.equity, mp.magic, mp.comment);
   }

   FileClose(handle);
}
